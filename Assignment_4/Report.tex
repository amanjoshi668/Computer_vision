    \hypertarget{grabcut}{%
\subsection{Grabcut}\label{grabcut}}

\begin{itemize}
\tightlist
\item
  We give a bounding box.
\item
  Assign GMM components to pixels
\item
  Learn GMM parameters from data
\item
  Make graph(n-links between pixels and t-links between source, sink and
  pixels)
\item
  Estimate segmentation: use min cut to solve
\item
  Repeat the process
\end{itemize}
